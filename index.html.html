function doPost(e) {
  var s = SpreadsheetApp.getActiveSheet();
  if (s.getLastRow()===0) s.appendRow(['Timestamp','Full Name','ID Type','ID Via',
    'Transport','Train/Flight No','Date of Arrival','Time of Arrival',
    'Pickup Needed','Pickup Point','Logistics Support']);
  var d = JSON.parse(e.postData.contents);
  s.appendRow([new Date(),d.fullName,d.idType,d.idVia,d.transport,
    d.flightTrain,d.arrDate,d.arrTime,d.pickup,d.pickupPoint,d.support]);
  return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
